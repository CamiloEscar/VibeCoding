 26    def __init__(self) -> None:
 27        """Initialize the factorial calculator with an empty cache."""
 28        self._cache: dict[int, int] = {0: 1, 1: 1}
 29
 30    def calculate(self, n: int | str) -> int:
 31        """
 32        Calculate the factorial of a given number.
 33
 34        This method uses an iterative approach with memoization for
 35        optimal performance. The result is cached for future use.
 36
 37        Args:
 38            n: A non-negative integer for which to calculate the factorial.
 39
 40        Returns:
 41            int: The factorial of n (n!).
 42
 43        Raises:
 44            InvalidInputError: If n is invalid or out of range.
 45            OverflowError: If calculation would cause overflow.
 46
 47        Examples:
 48            >>> calc = FactorialCalculator()
 49            >>> calc.calculate(5)
 50            120
 51            >>> calc.calculate(0)
 52            1
 53        """
 54        # Validate input
 55        n = InputValidator.validate_number(n)
 56
 57        # Check cache first
 58        if n in self._cache:
 59            return self._cache[n]
 60
 61        # Calculate factorial iteratively with overflow detection
 62        try:
 63            result = 1
 64            for i in range(2, n + 1):
 65                prev_result = result
 66                result *= i
 67
 68                # Check for potential overflow by verifying result integrity
 69                if result // i != prev_result:
 70                    raise OverflowError(
 71                        f"Factorial calculation for {n} would cause overflow"
 72                    )
 73
 74            # Cache the result
 75            self._cache[n] = result
 76            return result
 77
 78        except MemoryError as e:
 79            raise OverflowError(
 80                f"Factorial calculation for {n} exceeded memory limits"
 81            ) from e
 82
 83    def calculate_range(self, start: int | str, end: int | str) -> dict[int, int]:
 84        """
 85        Calculate factorials for a range of numbers.
 86
 87        Args:
 88            start: Starting number (inclusive).
 89            end: Ending number (inclusive).
 90
 91        Returns:
 92            Dict[int, int]: Dictionary mapping numbers to their factorials.
 93
 94        Raises:
 95            InvalidInputError: If range is invalid.
 96
 97        Examples:
 98            >>> calc = FactorialCalculator()
 99            >>> calc.calculate_range(3, 5)
100            {3: 6, 4: 24, 5: 120}
101        """
102        start = InputValidator.validate_number(start)
103        end = InputValidator.validate_number(end)
104
105        if start > end:
106            start, end = end, start
107
108        results = {}
109        for i in range(start, end + 1):
110            results[i] = self.calculate(i)
111
112        return results

115        """

121        self._cache = {0: 1, 1: 1}

124        """

129        """
130        return len(self._cache)
